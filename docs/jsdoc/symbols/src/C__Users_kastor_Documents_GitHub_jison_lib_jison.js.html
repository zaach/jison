115 generator.processGrammar = function processGrammarDef (grammar) {
116     var bnf = grammar.bnf,
117         tokens = grammar.tokens,
118         nonterminals = this.nonterminals = {},
119         productions = this.productions,
120         self = this;
121 
122     if (!grammar.bnf && grammar.ebnf) {
123         bnf = grammar.bnf = ebnfParser.transform(grammar.ebnf);
124     }
125 
126     if (tokens) {
127         if (typeof tokens === 'string') {
128             tokens = tokens.trim().split(' ');
129         } else {
130             tokens = tokens.slice(0);
131         }
132     }
133 
134     var symbols = this.symbols = [];
135 
136     // calculate precedence of operators
137     var operators = this.operators = processOperators(grammar.operators);
138 
139     // build productions from cfg
140     this.buildProductions(bnf, productions, nonterminals, symbols, operators);
141 
142     if (tokens && this.terminals.length !== tokens.length) {
143         self.trace("Warning: declared tokens differ from tokens found in rules.");
144         self.trace(this.terminals);
145         self.trace(tokens);
146     }
147 
148     // augment the grammar
149     this.augmentGrammar(grammar);
150 };

152 generator.augmentGrammar = function augmentGrammar (grammar) {
153     if (this.productions.length === 0) {
154         throw new Error("Grammar error: must have at least one rule.");
155     }
156     // use specified start symbol, or default to first user defined production
157     this.startSymbol = grammar.start || grammar.startSymbol || this.productions[0].symbol;
158     if (!this.nonterminals[this.startSymbol]) {
159         throw new Error("Grammar error: startSymbol must be a non-terminal found in your grammar.");
160     }
161     this.EOF = "$end";
162 
163     // augment the grammar
164     var acceptProduction = new Production('$accept', [this.startSymbol, '$end'], 0);
165     this.productions.unshift(acceptProduction);
166 
167     // prepend parser tokens
168     this.symbols.unshift("$accept",this.EOF);
169     this.symbols_.$accept = 0;
170     this.symbols_[this.EOF] = 1;
171     this.terminals.unshift(this.EOF);
172 
173     this.nonterminals.$accept = new Nonterminal("$accept");
174     this.nonterminals.$accept.productions.push(acceptProduction);
175 
176     // add follow $ to start symbol
177     this.nonterminals[this.startSymbol].follows.push(this.EOF);
178 };

180 // set precedence and associativity of operators
181 function processOperators (ops) {
182     if (!ops) return {};
183     var operators = {};
184     for (var i=0,k,prec;prec=ops[i]; i++) {
185         for (k=1;k < prec.length;k++) {
186             operators[prec[k]] = {precedence: i+1, assoc: prec[0]};
187         }
188     }
189     return operators;
190 }

193 generator.buildProductions = function buildProductions(bnf, productions, nonterminals, symbols, operators) {
194     var actions = [
195       '/* this == yyval */',
196       this.actionInclude || '',
197       'var $0 = $$.length - 1;',
198       'switch (yystate) {'
199     ];
200     var prods, symbol;
201     var productions_ = [0];
202     var symbolId = 1;
203     var symbols_ = {};
204 
205     var her = false; // has error recovery
206 
207     function addSymbol (s) {
208         if (s && !symbols_[s]) {
209             symbols_[s] = ++symbolId;
210             symbols.push(s);
211         }
212     }
213 
214     // add error symbol; will be third symbol, or "2" ($accept, $end, error)
215     addSymbol("error");
216 
217     for (symbol in bnf) {
218         if (!bnf.hasOwnProperty(symbol)) continue;
219 
220         addSymbol(symbol);
221         nonterminals[symbol] = new Nonterminal(symbol);
222 
223         if (typeof bnf[symbol] === 'string') {
224             prods = bnf[symbol].split(/\s*\|\s*/g);
225         } else {
226             prods = bnf[symbol].slice(0);
227         }
228 
229         prods.forEach(buildProduction);
230     }
231 
232     var sym, terms = [], terms_ = {};
233     each(symbols_, function (id, sym) {
234         if (!nonterminals[sym]) {
235             terms.push(sym);
236             terms_[id] = sym;
237         }
238     });
239 
240     this.hasErrorRecovery = her;
241 
242     this.terminals = terms;
243     this.terminals_ = terms_;
244     this.symbols_ = symbols_;
245 
246     this.productions_ = productions_;
247     actions.push('}');
248 
249     actions = actions.join("\n")
250                 .replace(/YYABORT/g, 'return false')
251                 .replace(/YYACCEPT/g, 'return true');
252 
253     var parameters = "yytext, yyleng, yylineno, yy, yystate /* action[1] */, $$ /* vstack */, _$ /* lstack */";
254     if (this.parseParams) parameters += ', ' + this.parseParams.join(', ');
255 
256     // first try to create the performAction function the old way,
257     // but this will break for some legal constructs in the user action code:
258     try {
259         this.performAction = Function(parameters, actions);
260     } catch (e) {
261         this.performAction = "function anonymous(" + parameters + ") {\n" + actions + "\n}";
262     }
263 
264     function buildProduction (handle) {
265         var r, rhs, i;
266         if (handle.constructor === Array) {
267             rhs = (typeof handle[0] === 'string') ?
268                       handle[0].trim().split(' ') :
269                       handle[0].slice(0);
270 
271             for (i=0; i<rhs.length; i++) {
272                 if (rhs[i] === 'error') her = true;
273                 if (!symbols_[rhs[i]]) {
274                     addSymbol(rhs[i]);
275                 }
276             }
277 
278             if (typeof handle[1] === 'string' || handle.length == 3) {
279                 // semantic action specified
280                 var action = 'case '+(productions.length+1)+':'+handle[1]+'\nbreak;';
281 
282                 // replace named semantic values ($nonterminal)
283                 if (action.match(/[$@][a-zA-Z][a-zA-Z0-9_]*/)) {
284                     var count = {},
285                         names = {};
286                     for (i=0;i<rhs.length;i++) {
287                         // check for aliased names, e.g., id[alias]
288                         var rhs_i = rhs[i].match(/\[[a-zA-Z][a-zA-Z0-9_-]*\]/);
289                         if (rhs_i) {
290                             rhs_i = rhs_i[0].substr(1, rhs_i[0].length-2);
291                             rhs[i] = rhs[i].substr(0, rhs[i].indexOf('['));
292                         } else {
293                             rhs_i = rhs[i];
294                         }
295 
296                         if (names[rhs_i]) {
297                             names[rhs_i + (++count[rhs_i])] = i+1;
298                         } else {
299                             names[rhs_i] = i+1;
300                             names[rhs_i + "1"] = i+1;
301                             count[rhs_i] = 1;
302                         }
303                     }
304                     action = action.replace(/\$([a-zA-Z][a-zA-Z0-9_]*)/g, function (str, pl) {
305                             return names[pl] ? '$'+names[pl] : str;
306                         }).replace(/@([a-zA-Z][a-zA-Z0-9_]*)/g, function (str, pl) {
307                             return names[pl] ? '@'+names[pl] : str;
308                         });
309                 }
310                 action = action
311                     // replace references to $$ with this.$, and @$ with this._$
312                     .replace(/([^'"])\$\$|^\$\$/g, '$1this.$').replace(/@[0$]/g, "this._$")
313 
314                     // replace semantic value references ($n) with stack value (stack[n])
315                     .replace(/\$(-?\d+)/g, function (_, n) {
316                         return "$$[$0" + (parseInt(n, 10) - rhs.length || '') + "]";
317                     })
318                     // same as above for location references (@n)
319                     .replace(/@(-?\d+)/g, function (_, n) {
320                         return "_$[$0" + (n - rhs.length || '') + "]";
321                     });
322                 actions.push(action);
323 
324                 // done with aliases; strip them.
325                 rhs = rhs.map(function(e,i) { return e.replace(/\[[a-zA-Z_][a-zA-Z0-9_-]*\]/g, '') });
326                 r = new Production(symbol, rhs, productions.length+1);
327                 // precedence specified also
328                 if (handle[2] && operators[handle[2].prec]) {
329                     r.precedence = operators[handle[2].prec].precedence;
330                 }
331             } else {
332                 // no action -> don't care about aliases; strip them.
333                 rhs = rhs.map(function(e,i) { return e.replace(/\[[a-zA-Z_][a-zA-Z0-9_-]*\]/g, '') });
334                 // only precedence specified
335                 r = new Production(symbol, rhs, productions.length+1);
336                 if (operators[handle[1].prec]) {
337                     r.precedence = operators[handle[1].prec].precedence;
338                 }
339             }
340         } else {
341             // no action -> don't care about aliases; strip them.
342             handle = handle.replace(/\[[a-zA-Z_][a-zA-Z0-9_-]*\]/g, '');
343             rhs = handle.trim().split(' ');
344             for (i=0; i<rhs.length; i++) {
345                 if (rhs[i] === 'error') her = true;
346                 if (!symbols_[rhs[i]]) {
347                     addSymbol(rhs[i]);
348                 }
349             }
350             r = new Production(symbol, rhs, productions.length+1);
351         }
352         if (r.precedence === 0) {
353             // set precedence
354             for (i=r.handle.length-1; i>=0; i--) {
355                 if (!(r.handle[i] in nonterminals) && r.handle[i] in operators) {
356                     r.precedence = operators[r.handle[i]].precedence;
357                 }
358             }
359         }
360 
361         productions.push(r);
362         productions_.push([symbols_[r.symbol], r.handle[0] === '' ? 0 : r.handle.length]);
363         nonterminals[symbol].productions.push(r);
364     }
365 };

418 // calculate follow sets typald on first and nullable
419 lookaheadMixin.followSets = function followSets () {
420     var productions = this.productions,
421         nonterminals = this.nonterminals,
422         self = this,
423         cont = true;
424 
425     // loop until no further changes have been made
426     while(cont) {
427         cont = false;
428 
429         productions.forEach(function Follow_prod_forEach (production, k) {
430             //self.trace(production.symbol,nonterminals[production.symbol].follows);
431             // q is used in Simple LALR algorithm determine follows in context
432             var q;
433             var ctx = !!self.go_;
434 
435             var set = [],oldcount;
436             for (var i=0,t;t=production.handle[i];++i) {
437                 if (!nonterminals[t]) continue;
438 
439                 // for Simple LALR algorithm, self.go_ checks if
440                 if (ctx)
441                     q = self.go_(production.symbol, production.handle.slice(0, i));
442                 var bool = !ctx || q === parseInt(self.nterms_[t], 10);
443 
444                 if (i === production.handle.length+1 && bool) {
445                     set = nonterminals[production.symbol].follows;
446                 } else {
447                     var part = production.handle.slice(i+1);
448 
449                     set = self.first(part);
450                     if (self.nullable(part) && bool) {
451                         set.push.apply(set, nonterminals[production.symbol].follows);
452                     }
453                 }
454                 oldcount = nonterminals[t].follows.length;
455                 Set.union(nonterminals[t].follows, set);
456                 if (oldcount !== nonterminals[t].follows.length) {
457                     cont = true;
458                 }
459             }
460         });
461     }

464 // return the FIRST set of a symbol or series of symbols
465 lookaheadMixin.first = function first (symbol) {
466     // epsilon
467     if (symbol === '') {
468         return [];
469     // RHS
470     } else if (symbol instanceof Array) {
471         var firsts = [];
472         for (var i=0,t;t=symbol[i];++i) {
473             if (!this.nonterminals[t]) {
474                 if (firsts.indexOf(t) === -1)
475                     firsts.push(t);
476             } else {
477                 Set.union(firsts, this.nonterminals[t].first);
478             }
479             if (!this.nullable(t))
480                 break;
481         }
482         return firsts;
483     // terminal
484     } else if (!this.nonterminals[symbol]) {
485         return [symbol];
486     // nonterminal
487     } else {
488         return this.nonterminals[symbol].first;
489     }
490 };

525 // fixed-point calculation of NULLABLE
526 lookaheadMixin.nullableSets = function nullableSets () {
527     var firsts = this.firsts = {},
528         nonterminals = this.nonterminals,
529         self = this,
530         cont = true;
531 
532     // loop until no further changes have been made
533     while(cont) {
534         cont = false;
535 
536         // check if each production is nullable
537         this.productions.forEach(function (production, k) {
538             if (!production.nullable) {
539                 for (var i=0,n=0,t;t=production.handle[i];++i) {
540                     if (self.nullable(t)) n++;
541                 }
542                 if (n===i) { // production is nullable if all tokens are nullable
543                     production.nullable = cont = true;
544                 }
545             }
546         });
547 
548         //check if each symbol is nullable
549         for (var symbol in nonterminals) {
550             if (!this.nullable(symbol)) {
551                 for (var i=0,production;production=nonterminals[symbol].productions.item(i);i++) {
552                     if (production.nullable)
553                         nonterminals[symbol].nullable = cont = true;
554                 }
555             }
556         }
557     }

560 // check if a token or series of tokens is nullable
561 lookaheadMixin.nullable = function nullable (symbol) {
562     // epsilon
563     if (symbol === '') {
564         return true;
565     // RHS
566     } else if (symbol instanceof Array) {
567         for (var i=0,t;t=symbol[i];++i) {
568             if (!this.nullable(t))
569                 return false;
570         }
571         return true;
572     // terminal
573     } else if (!this.nonterminals[symbol]) {
574         return false;
575     // nonterminal
576     } else {
577         return this.nonterminals[symbol].nullable;
578     }

676 lrGeneratorMixin.closureOperation = function closureOperation (itemSet /*, closureSet*/) {
677     var closureSet = new this.ItemSet();
678     var self = this;
679 
680     var set = itemSet,
681         itemQueue, syms = {};
682 
683     do {
684     itemQueue = new Set();
685     closureSet.concat(set);
686     set.forEach(function CO_set_forEach (item) {
687         var symbol = item.markedSymbol;
688 
689         // if token is a non-terminal, recursively add closures
690         if (symbol && self.nonterminals[symbol]) {
691             if(!syms[symbol]) {
692                 self.nonterminals[symbol].productions.forEach(function CO_nt_forEach (production) {
693                     var newItem = new self.Item(production, 0);
694                     if(!closureSet.contains(newItem))
695                         itemQueue.push(newItem);
696                 });
697                 syms[symbol] = true;
698             }
699         } else if (!symbol) {
700             // reduction
701             closureSet.reductions.push(item);
702             closureSet.inadequate = closureSet.reductions.length > 1 || closureSet.shifts;
703         } else {
704             // shift
705             closureSet.shifts = true;
706             closureSet.inadequate = closureSet.reductions.length > 0;
707         }

776 lrGeneratorMixin.parseTable = function parseTable (itemSets) {
777     var states = [],
778         nonterminals = this.nonterminals,
779         operators = this.operators,
780         conflictedStates = {}, // array of [state, token] tuples
781         self = this,
782         s = 1, // shift
783         r = 2, // reduce
784         a = 3; // accept
785 
786     // for each item set
787     itemSets.forEach(function (itemSet, k) {
788         var state = states[k] = {};
789         var action, stackSymbol;
790 
791         // set shift and goto actions
792         for (stackSymbol in itemSet.edges) {
793             itemSet.forEach(function (item, j) {
794                 // find shift and goto actions
795                 if (item.markedSymbol == stackSymbol) {
796                     var gotoState = itemSet.edges[stackSymbol];
797                     if (nonterminals[stackSymbol]) {
798                         // store state to go to after a reduce
799                         //self.trace(k, stackSymbol, 'g'+gotoState);
800                         state[self.symbols_[stackSymbol]] = gotoState;
801                     } else {
802                         //self.trace(k, stackSymbol, 's'+gotoState);
803                         state[self.symbols_[stackSymbol]] = [s,gotoState];
804                     }
805                 }
806             });
807         }
808 
809         // set accept action
810         itemSet.forEach(function (item, j) {
811             if (item.markedSymbol == self.EOF) {
812                 // accept
813                 state[self.symbols_[self.EOF]] = [a];
814                 //self.trace(k, self.EOF, state[self.EOF]);
815             }
816         });
817 
818         var allterms = self.lookAheads ? false : self.terminals;
819 
820         // set reductions and resolve potential conflicts
821         itemSet.reductions.forEach(function (item, j) {
822             // if parser uses lookahead, only enumerate those terminals
823             var terminals = allterms || self.lookAheads(itemSet, item);
824 
825             terminals.forEach(function (stackSymbol) {
826                 action = state[self.symbols_[stackSymbol]];
827                 var op = operators[stackSymbol];
828 
829                 // Reading a terminal and current position is at the end of a production, try to reduce
830                 if (action || action && action.length) {
831                     var sol = resolveConflict(item.production, op, [r,item.production.id], action[0] instanceof Array ? action[0] : action);
832                     self.resolutions.push([k,stackSymbol,sol]);
833                     if (sol.bydefault) {
834                         self.conflicts++;
835                         if (!self.DEBUG) {
836                             self.warn('Conflict in grammar: multiple actions possible when lookahead token is ',stackSymbol,' in state ',k, "\n- ", printAction(sol.r, self), "\n- ", printAction(sol.s, self));
837                             conflictedStates[k] = true;
838                         }
839                         if (self.options.noDefaultResolve) {
840                             if (!(action[0] instanceof Array))
841                                 action = [action];
842                             action.push(sol.r);
843                         }
844                     } else {
845                         action = sol.action;
846                     }
847                 } else {
848                     action = [r,item.production.id];
849                 }
850                 if (action && action.length) {
851                     state[self.symbols_[stackSymbol]] = action;
852                 } else if (action === NONASSOC) {
853                     state[self.symbols_[stackSymbol]] = undefined;
854                 }
855             });
856         });
857 
858     });
859 
860     if (!self.DEBUG && self.conflicts > 0) {
861         self.warn("\nStates with conflicts:");
862         each(conflictedStates, function (val, state) {
863             self.warn('State '+state);
864             self.warn('  ',itemSets.item(state).join("\n  "));
865         });
866     }
867 
868     return states;
869 };

889 // resolves shift-reduce and reduce-reduce conflicts
890 function resolveConflict (production, op, reduce, shift) {
891     var sln = {production: production, operator: op, r: reduce, s: shift},
892         s = 1, // shift
893         r = 2, // reduce
894         a = 3; // accept
895 
896     if (shift[0] === r) {
897         sln.msg = "Resolve R/R conflict (use first production declared in grammar.)";
898         sln.action = shift[1] < reduce[1] ? shift : reduce;
899         if (shift[1] !== reduce[1]) sln.bydefault = true;
900         return sln;
901     }
902 
903     if (production.precedence === 0 || !op) {
904         sln.msg = "Resolve S/R conflict (shift by default.)";
905         sln.bydefault = true;
906         sln.action = shift;
907     } else if (production.precedence < op.precedence ) {
908         sln.msg = "Resolve S/R conflict (shift for higher precedent operator.)";
909         sln.action = shift;
910     } else if (production.precedence === op.precedence) {
911         if (op.assoc === "right" ) {
912             sln.msg = "Resolve S/R conflict (shift for right associative operator.)";
913             sln.action = shift;
914         } else if (op.assoc === "left" ) {
915             sln.msg = "Resolve S/R conflict (reduce for left associative operator.)";
916             sln.action = reduce;
917         } else if (op.assoc === "nonassoc" ) {
918             sln.msg = "Resolve S/R conflict (no action for non-associative operator.)";
919             sln.action = NONASSOC;
920         }
921     } else {
922         sln.msg = "Resolve conflict (reduce for higher precedent production.)";
923         sln.action = reduce;
924     }
925 
926     return sln;
927 }

1255     function lex() {
1256         var token;
1257         token = self.lexer.lex() || EOF; // $end = 1
1258         // if token isn't its numeric value, convert
1259         if (typeof token !== 'number') {
1260             token = self.symbols_[token] || token;
1261         }

1265     var symbol, preErrorSymbol, state, action, a, r, yyval = {}, p, len, newState, expected;
1266     while (true) {

1268         state = stack[stack.length - 1];

1275                 symbol = lex();

1279         }

1289             function locateNearestErrorRecoveryRule(state) {
1290                 var stack_probe = stack.length - 1;
1291                 var depth = 0;
1292

1298                     }
1299                     if (state === 0 || stack_probe < 2) {
1300                         return false; // No suitable error recovery rule available.
1301                     }

1303                     state = stack[stack_probe];
1304                     ++depth;
1305                 }
1306             }

1322                     errStr = 'Parse error on line '+(yylineno+1)+": Unexpected " +
1323                                   (symbol == EOF ? "end of input" :
1324                                               ("'"+(this.terminals_[symbol] || symbol)+"'"));
1325                 }
1326                 this.parseError(errStr, {
1327                     text: this.lexer.match,
1328                     token: this.terminals_[symbol] || symbol,
1329                     line: this.lexer.yylineno,
1330                     loc: yyloc,
1331                     expected: expected,
1332                     recoverable: (error_rule_depth !== false)
1333                 });
1334             } else if (preErrorSymbol !== EOF) {
1335                 error_rule_depth = locateNearestErrorRecoveryRule(state);
1336             }
1337 
1338             // just recovered from another error
1339             if (recovering == 3) {
1340                 if (symbol === EOF || preErrorSymbol === EOF) {
1341                     throw new Error(errStr || 'Parsing halted while starting to recover from another error.');
1342                 }
1343 
1344                 // discard current lookahead and grab another
1345                 yyleng = this.lexer.yyleng;
1346                 yytext = this.lexer.yytext;
1347                 yylineno = this.lexer.yylineno;
1348                 yyloc = this.lexer.yylloc;
1349                 symbol = lex();
1350             }
1351 
1352             // try to recover from error
1353             if (error_rule_depth === false) {
1354                 throw new Error(errStr || 'Parsing halted. No suitable error recovery rule available.');
1355             }
1356             popStack(error_rule_depth);
1357 
1358             preErrorSymbol = (symbol == TERROR ? null : symbol); // save the lookahead token
1359             symbol = TERROR;         // insert generic error symbol as new lookahead
1360             state = stack[stack.length-1];
1361             action = table[state] && table[state][TERROR];
1362             recovering = 3; // allow 3 real symbols to be shifted before reporting a new error
1363         }
1364 
1365         // this shouldn't happen, unless resolve defaults are off
1366         if (action[0] instanceof Array && action.length > 1) {
1367             throw new Error('Parse Error: multiple actions possible at state: '+state+', token: '+symbol);
1368         }
1369 
1370         switch (action[0]) {
1371             case 1: // shift
1372                 //this.shiftCount++;
1373 
1374                 stack.push(symbol);
1375                 vstack.push(this.lexer.yytext);
1376                 lstack.push(this.lexer.yylloc);
1377                 stack.push(action[1]); // push state
1378                 symbol = null;
1379                 if (!preErrorSymbol) { // normal execution/no error
1380                     yyleng = this.lexer.yyleng;
1381                     yytext = this.lexer.yytext;
1382                     yylineno = this.lexer.yylineno;
1383                     yyloc = this.lexer.yylloc;
1384                     if (recovering > 0) {
1385                         recovering--;
1386                     }
1387                 } else {
1388                     // error just occurred, resume old lookahead f/ before error
1389                     symbol = preErrorSymbol;
1390                     preErrorSymbol = null;
1391                 }
1392                 break;
1393 
1394             case 2:
1395                 // reduce
1396                 //this.reductionCount++;
1397 
1398                 len = this.productions_[action[1]][1];
1399 
1400                 // perform semantic action
1401                 yyval.$ = vstack[vstack.length-len]; // default to $$ = $1
1402                 // default location, uses first token for firsts, last for lasts
1403                 yyval._$ = {
1404                     first_line: lstack[lstack.length-(len||1)].first_line,
1405                     last_line: lstack[lstack.length-1].last_line,
1406                     first_column: lstack[lstack.length-(len||1)].first_column,
1407                     last_column: lstack[lstack.length-1].last_column
1408                 };
1409                 if (ranges) {
1410                   yyval._$.range = [lstack[lstack.length-(len||1)].range[0], lstack[lstack.length-1].range[1]];
1411                 }
1412                 r = this.performAction.apply(yyval, [yytext, yyleng, yylineno, this.yy, action[1], vstack, lstack].concat(args));
1413 
1414                 if (typeof r !== 'undefined') {
1415                     return r;
1416                 }
1417 
1418                 // pop off stack
1419                 if (len) {
1420                     stack = stack.slice(0,-1*len*2);
1421                     vstack = vstack.slice(0, -1*len);
1422                     lstack = lstack.slice(0, -1*len);
1423                 }
1424 
1425                 stack.push(this.productions_[action[1]][0]);    // push nonterminal (reduce)
1426                 vstack.push(yyval.$);
1427                 lstack.push(yyval._$);
1428                 // goto new state = table[STATE][NONTERMINAL]
1429                 newState = table[stack[stack.length-2]][stack[stack.length-1]];
1430                 stack.push(newState);
1431                 break;
1432 
1433             case 3:
1434                 // accept
1435                 return true;
1436         }
1437 
1438     }
1439 
1440     return true;
1441 };
